}

char* strncat(char* s, const char* t, size_t n) { // (4 pts) (7)
char* p = s;
while(*s != '\0') { ++s; }
while(n-- > 0 && *t != '\0') { *s++ = *t++; }
*s = '\0';
return p;
}

int strcmp(const char* s, const char* t) { // (4 pts) (8)
while(*s != '\0' && *t != '\0' && *s == *t) {
++s;
++t;
}
return *s - *t;
}
